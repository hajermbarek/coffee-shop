function testEventForm() {
    let bouton = document.getElementById("btnTester");
    let formulaire = document.getElementById("formulaire");

    bouton.addEventListener("click", function(e) {
        e.preventDefault(); // Empêche la soumission du formulaire

        // Récupère les valeurs des champs
        let nom = document.getElementById("nom").value;
        let email = document.getElementById("email").value;

        // Vérifie si les champs sont remplis
        if (nom === "" || email === "") {
            alert("Veuillez remplir tous les champs !");
            return; // stoppe la fonction si des champs sont vides
        }

        // Affiche les valeurs dans la console
        console.log("Nom :", nom);
        console.log("Email :", email);
        console.log("Événement déclenché par :", e.target);

        // Confirmation à l'utilisateur
        alert(`Bonjour ${nom}, votre email (${email}) a été testé avec succès !`);
    });
}

// Appelle la fonction pour attacher l'événement
testEventForm();
