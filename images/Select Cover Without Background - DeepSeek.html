// ---------- HELPER FUNCTIONS ----------
function generateReservationCode() {
    return Math.floor(100000 + Math.random() * 900000).toString();
}

function getUserId() {
    let userId = localStorage.getItem("cozyCafe_userId");
    if (!userId) {
        userId = "user_" + Date.now() + "_" + Math.random().toString(36).substr(2, 9);
        localStorage.setItem("cozyCafe_userId", userId);
    }
    return userId;
}

// Get all book reservations from storage
function getBookReservations() {
    const reservations = localStorage.getItem("cozyCafe_bookReservations");
    return reservations ? JSON.parse(reservations) : [];
}

// Save all book reservations to storage
function saveBookReservations(reservations) {
    localStorage.setItem("cozyCafe_bookReservations", JSON.stringify(reservations));
    window.dispatchEvent(new Event("bookReservationsUpdated"));
}

// Check and clean expired book reservations (older than 7 days)
function cleanExpiredBookReservations() {
    const reservations = getBookReservations();
    const now = new Date();
    let changed = false;
    
    const activeReservations = reservations.filter(res => {
        const expiryDate = new Date(res.reservedUntil);
        if (expiryDate < now) {
            changed = true;
            return false;
        }
        return true;
    });
    
    if (changed) {
        saveBookReservations(activeReservations);
    }
    return activeReservations;
}

// Get user's current active book reservation
function getUserBookReservation() {
    cleanExpiredBookReservations();
    const userId = getUserId();
    const reservations = getBookReservations();
    return reservations.find(res => res.userId === userId);
}

// Check if a book is reserved by anyone
function isBookReserved(bookId) {
    cleanExpiredBookReservations();
    const reservations = getBookReservations();
    return reservations.some(res => res.bookId === bookId);
}

// Check if a book is reserved by current user
function isBookReservedByUser(bookId) {
    const userId = getUserId();
    const reservations = getBookReservations();
    return reservations.some(res => res.bookId === bookId && res.userId === userId);
}

// Reserve a book
function reserveBook(bookId, bookTitle) {
    // Check if user already has a reservation
    const existingReservation = getUserBookReservation();
    if (existingReservation) {
        const bookInfo = booksDatabase.find(b => b.id === existingReservation.bookId);
        alert(`You already have "${bookInfo?.title || existingReservation.bookId}" reserved until ${new Date(existingReservation.reservedUntil).toLocaleDateString()}. You can only reserve one book at a time.`);
        return null;
    }
    
    // Check if book is already reserved by someone else
    if (isBookReserved(bookId)) {
        alert(`Sorry, "${bookTitle}" is already reserved by another reader. Please choose another book.`);
        return null;
    }
    
    // Create new reservation
    const code = generateReservationCode();
    const reservedUntil = new Date();
    reservedUntil.setDate(reservedUntil.getDate() + 7);
    
    const newReservation = {
        bookId: bookId,
        bookTitle: bookTitle,
        userId: getUserId(),
        code: code,
        reservedAt: new Date().toISOString(),
        reservedUntil: reservedUntil.toISOString()
    };
    
    const reservations = getBookReservations();
    reservations.push(newReservation);
    saveBookReservations(reservations);
    
    // Store the code separately for easy access on reservation page
    localStorage.setItem("cozyCafe_activeBookCode", code);
    localStorage.setItem("cozyCafe_activeBookTitle", bookTitle);
    
    return newReservation;
}

// Cancel book reservation
function cancelBookReservation(bookId) {
    const userId = getUserId();
    let reservations = getBookReservations();
    reservations = reservations.filter(res => !(res.bookId === bookId && res.userId === userId));
    saveBookReservations(reservations);
    
    // Clear stored code if it was this user's
    const userRes = getUserBookReservation();
    if (!userRes) {
        localStorage.removeItem("cozyCafe_activeBookCode");
        localStorage.removeItem("cozyCafe_activeBookTitle");
    }
    return true;
}

// Verify book reservation code
function verifyBookReservationCode(enteredCode) {
    cleanExpiredBookReservations();
    const reservations = getBookReservations();
    const matchingReservation = reservations.find(res => res.code === enteredCode);
    
    if (matchingReservation) {
        return {
            valid: true,
            reservation: matchingReservation
        };
    }
    return {
        valid: false,
        message: "Invalid code. Please check your reservation code."
    };
}

function updateBooksReservationStatus() {
    const bookCards = document.querySelectorAll(".book");

        if (!bookNameElement) return;

            const isReservedByUser = isBookReservedByUser(bookInfo.id);
            
            const existingBadge = card.querySelector(".reservation-status");
            if (existingBadge) existingBadge.remove();
            
            const badge = document.createElement("div");
            badge.className = "reservation-status";

                card.classList.add("user-reserved");

                    moreLink.style.pointerEvents = "";
